📋 Instruction Sum Type

data Instruction
  -- ── RV64I 基礎 ──────────────────
  = ADD    Register Register Register  -- rd = rs1 + rs2
  | ADDI   Register Register Imm12     -- rd = rs1 + sext(imm)
  | LUI    Register Imm20              -- rd = imm << 12
  | BEQ    Register Register Imm13     -- branch if equal
  | JAL    Register Imm21              -- jump and link
  | CSRRW  Register CSRAddr Register   -- CSR read/write
  -- ── RV64M 乘除法 ────────────────
  | MUL    Register Register Register
  | DIV    Register Register Register
  -- ── Privileged ──────────────────
  | MRET                               -- return from M-mode
  | SFENCE_VMA Register Register       -- TLB flush
  deriving (Show, Eq, Ord, Generic)

Phase 1Encode / Decode 雙向轉換

encode :: Instruction → Word32

encode :: Instruction -> Word32
encode = \case
  -- R-type: [funct7|rs2|rs1|f3|rd|op]
  ADD  rd rs1 rs2 -> buildR 0x33 (r rd) 0x0 (r rs1) (r rs2) 0x00
  SUB  rd rs1 rs2 -> buildR 0x33 (r rd) 0x0 (r rs1) (r rs2) 0x20
  MUL  rd rs1 rs2 -> buildR 0x33 (r rd) 0x0 (r rs1) (r rs2) 0x01
  -- I-type: [imm12|rs1|f3|rd|op]
  ADDI rd rs1 imm -> buildI 0x13 (r rd) 0x0 (r rs1) (i12 imm)
  -- B-type: [imm13 encoded|rs2|rs1|f3|op]
  BEQ  rs1 rs2 imm -> buildB 0x63 0x0 (r rs1) (r rs2) (i13 imm)
  -- J-type
  JAL  rd imm -> buildJ 0x6F (r rd) (i21 imm)
  MRET        -> buildR 0x73 0 0 0 2 0x18  -- spec §3.3.2

decode :: Word32 → Either DecodeError Instruction

decode :: Word32 -> Either DecodeError Instruction
decode w = case opcode w of
  0x33 -> decodeR33 w
  0x13 -> decodeI13 w
  0x63 -> decodeBranch w
  0x73 -> decodeSystem w
  op   -> Left (UnknownOpcode op)

ConstraintDef 資料結構

data ConstraintDef = ConstraintDef

  , ctags      :: [Tag]           -- [Memory, Atomic, ...]

  , symRs1    :: SWord8

常用 Constraint 範例

rdNotZero :: ConstraintDef

alignedAddr8 = ConstraintDef "8B-align" [Memory]
  (\s -> (symRs1 s + sext (symImm s)) `sMod` 8 .== 0)

      bandit.update(JSON.parse(e.data))

  onMounted(connect)
  onUnmounted(() => es?.close())

③ Either 錯誤處理 Error Handling with Either

讓錯誤成為型別的一部分，強制 caller 處理 ／ Make failures visible in the type signature

-- ❌ Exception 的問題：
--   1. 型別簽名看不出可能失敗
--   2. Caller 可能忘記 catch
--   3. 純函數裡難以測試
decode :: Word32 -> Instruction  -- 隱藏失敗

-- ✅ Either 的優點：
decode :: Word32 -> Either DecodeError Instruction

data DecodeError
  = UnknownOpcode    Word32
  | UnknownFunct3    Word32 Word32
  | ReservedEncoding Word32

-- GHC 強制你處理 Left case：
processInstruction :: Word32 -> IO ()
processInstruction w = case decode w of
  Right instr -> execute instr
  Left (UnknownOpcode op) ->
    logWarning ("bad opcode: 0x" <> showHex op "")
  Left (ReservedEncoding raw) ->
    raiseIllegalInstruction raw

Monad 組合 Monadic Chaining

-- do-notation 讓 Either 組合很自然
-- 任何一步 Left 就短路，不繼續執行
processWord :: Word32 -> Either Error Result
processWord w = do
  instr  <- decode w              -- ① decode
  params <- extractParams instr   -- ② validate
  result <- process params        -- ③ execute
  return result

-- 等價於：
processWord w =
  decode w >>= \instr ->
  extractParams instr >>= \params ->
  process params >>= \result ->
  Right result
拇指原則 ／ Rule of thumb

預期可能發生 → Either / Maybe。真正的程式錯誤 → error。Expected = types; Unexpected = exceptions.

把 Haskell 值「提升」到 symbolic 層面，讓 Z3 替你求解 ／ Lift values to symbolic domain; let Z3 do the math

兩個世界的對應 Two Worlds

-- Symbolic monad：建立 SMT query
buildQuery :: Symbolic ()
buildQuery = do
  rd  <- sWord8 "rd"    -- 宣告 symbolic variable
  imm <- sInt32 "imm"

  -- constrain = 加入 Z3 assertion
  constrain $ rd  .< 32      -- 0 ≤ rd < 32
  constrain $ rd  ./= 0      -- rd ≠ x0
  constrain $ imm .>= (-2048)
  constrain $ imm .< 2048

result <- sat buildQuery
-- result 包含 rd, imm 的一個具體值

Constraint predicate 的魔法

-- 這個函數看起來像普通 Haskell
-- 但它建立的是 SMT assertion
rdNotZero :: SymInstrParams -> SBool
rdNotZero params =
  symRd params ./= 0
  -- 不是 Bool 比較！
  -- 是向 Z3 聲明 "rd 不等於 0 這個約束"

-- 8-byte aligned address constraint
alignedAddr8 :: SymInstrParams -> SBool
alignedAddr8 params =
  let addr = symRs1 params + sext (symImm params)
  in  addr `sMod` 8 .== 0
  -- sext = sign extend（symbolic version）
  -- sMod = symbolic modulo
精隨 ／ Key Insight

SBV 用 Haskell 的型別系統防止你在 symbolic 和 concrete 之間混用。SWord8 和 Word8 是不同型別。The type system separates the two worlds.

⑥ Hedgehog Property-Based Testing

Unit Test vs Property Test

  encode (ADD x1 x2 x3) @?= 0x003100B3

prop_roundtrip :: Property
prop_roundtrip = property $ do
  instr <- forAll genSampleInstruction
  -- Hedgehog 自動生成 100+ 個隨機 instr
  -- 任何一個失敗都是 bug
